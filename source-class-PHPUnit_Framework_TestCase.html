 715:      * Runs the bare test sequence.
 716:      */
 717:     public function runBare()
 718:     {
 719:         $this->numAssertions = 0;
 720: 
 721:         $this->snapshotGlobalState();
 722:         $this->startOutputBuffering();
 723:         clearstatcache();
 724:         $currentWorkingDirectory = getcwd();
 725: 
 726:         $hookMethods = PHPUnit_Util_Test::getHookMethods(get_class($this));
 727: 
 728:         try {
 729:             $hasMetRequirements = false;
 730:             $this->checkRequirements();
 731:             $hasMetRequirements = true;
 732: 
 733:             if ($this->inIsolation) {
 734:                 foreach ($hookMethods['beforeClass'] as $method) {
 735:                     $this->$method();
 736:                 }
 737:             }
 738: 
 739:             $this->setExpectedExceptionFromAnnotation();
 740: 
 741:             foreach ($hookMethods['before'] as $method) {
 742:                 $this->$method();
 743:             }
 744: 
 745:             $this->assertPreConditions();
 746:             $this->testResult = $this->runTest();
 747:             $this->verifyMockObjects();
 748:             $this->assertPostConditions();
 749: 
 750:             $this->status = PHPUnit_Runner_BaseTestRunner::STATUS_PASSED;
 751:         } catch (PHPUnit_Framework_IncompleteTest $e) {
 752:             $this->status        = PHPUnit_Runner_BaseTestRunner::STATUS_INCOMPLETE;
 753:             $this->statusMessage = $e->getMessage();
 754:         } catch (PHPUnit_Framework_SkippedTest $e) {
 755:             $this->status        = PHPUnit_Runner_BaseTestRunner::STATUS_SKIPPED;
 756:             $this->statusMessage = $e->getMessage();
 757:         } catch (PHPUnit_Framework_AssertionFailedError $e) {
 758:             $this->status        = PHPUnit_Runner_BaseTestRunner::STATUS_FAILURE;
 759:             $this->statusMessage = $e->getMessage();
 760:         } catch (PredictionException $e) {
 761:             $this->status        = PHPUnit_Runner_BaseTestRunner::STATUS_FAILURE;
 762:             $this->statusMessage = $e->getMessage();
 763:         } catch (Throwable $_e) {
 764:             $e = $_e;
 765:         } catch (Exception $_e) {
 766:             $e = $_e;
 767:         }
 768: 
 769:         if (isset($e)) {
 770:             $this->status        = PHPUnit_Runner_BaseTestRunner::STATUS_ERROR;
 771:             $this->statusMessage = $e->getMessage();
 772:         }
 773: 
 774:         // Clean up the mock objects.
 775:         $this->mockObjects = array();
 776:         $this->prophet     = null;
 777: 
 778:         // Tear down the fixture. An exception raised in tearDown() will be
 779:         // caught and passed on when no exception was raised before.
 780:         try {
 781:             if ($hasMetRequirements) {
 782:                 foreach ($hookMethods['after'] as $method) {
 783:                     $this->$method();
 784:                 }
 785: 
 786:                 if ($this->inIsolation) {
 787:                     foreach ($hookMethods['afterClass'] as $method) {
 788:                         $this->$method();
 789:                     }
 790:                 }
 791:             }
 792:         } catch (Throwable $_e) {
 793:             if (!isset($e)) {
 794:                 $e = $_e;
 795:             }
 796:         } catch (Exception $_e) {
 797:             if (!isset($e)) {
 798:                 $e = $_e;
 799:             }
 800:         }
 801: 
 802:         try {
 803:             $this->stopOutputBuffering();
 804:         } catch (PHPUnit_Framework_RiskyTestError $_e) {
 805:             if (!isset($e)) {
 806:                 $e = $_e;
 807:             }
 808:         }
 809: 
 810:         clearstatcache();
 811: 
 812:         if ($currentWorkingDirectory != getcwd()) {
 813:             chdir($currentWorkingDirectory);
 814:         }
 815: 
 816:         $this->restoreGlobalState();
 817: 
 818:         // Clean up INI settings.
 819:         foreach ($this->iniSettings as $varName => $oldValue) {
 820:             ini_set($varName, $oldValue);
 821:         }
 822: 
 823:         $this->iniSettings = array();

 855:     /**
 856:      * Override to run the test and assert its state.
 857:      *
 858:      * @return mixed
 859:      * @throws Exception|PHPUnit_Framework_Exception
 860:      * @throws PHPUnit_Framework_Exception
 861:      */
 862:     protected function runTest()
 863:     {
 864:         if ($this->name === null) {
 865:             throw new PHPUnit_Framework_Exception(
 866:                 'PHPUnit_Framework_TestCase::$name must not be null.'
 867:             );
 868:         }
 869: 
 870:         try {
 871:             $class  = new ReflectionClass($this);
 872:             $method = $class->getMethod($this->name);
 873:         } catch (ReflectionException $e) {
 874:             $this->fail($e->getMessage());
 875:         }
 876: 
 877:         try {
 878:             $testResult = $method->invokeArgs(
 879:                 $this,
 880:                 array_merge($this->data, $this->dependencyInput)
 881:             );
 882:         } catch (Throwable $_e) {
 883:             $e = $_e;
 884:         } catch (Exception $_e) {
 885:             $e = $_e;
 886:         }
 887: 
 888:         if (isset($e)) {
 889:             $checkException = false;
 890: 
 891:             if (is_string($this->expectedException)) {
 892:                 $checkException = true;
 893: 
 894:                 if ($e instanceof PHPUnit_Framework_Exception) {
 895:                     $checkException = false;
 896:                 }
 897: 
 898:                 $reflector = new ReflectionClass($this->expectedException);
 899: 
 900:                 if ($this->expectedException == 'PHPUnit_Framework_Exception' ||
 901:                     $reflector->isSubclassOf('PHPUnit_Framework_Exception')) {
 902:                     $checkException = true;
 903:                 }
 904:             }
 905: 
 906:             if ($checkException) {
 907:                 $this->assertThat(
 908:                     $e,
 909:                     new PHPUnit_Framework_Constraint_Exception(
 910:                         $this->expectedException
 911:                     )
 912:                 );
 913: 
 914:                 if (is_string($this->expectedExceptionMessage) &&
 915:                     !empty($this->expectedExceptionMessage)) {
 916:                     $this->assertThat(
 917:                         $e,
 918:                         new PHPUnit_Framework_Constraint_ExceptionMessage(
 919:                             $this->expectedExceptionMessage
 920:                         )
 921:                     );
 922:                 }
 923: 
 924:                 if (is_string($this->expectedExceptionMessageRegExp) &&
 925:                     !empty($this->expectedExceptionMessageRegExp)) {
 926:                     $this->assertThat(
 927:                         $e,
 928:                         new PHPUnit_Framework_Constraint_ExceptionMessageRegExp(
 929:                             $this->expectedExceptionMessageRegExp
 930:                         )
 931:                     );
 932:                 }
 933: 
 934:                 if ($this->expectedExceptionCode !== null) {
 935:                     $this->assertThat(
 936:                         $e,
 937:                         new PHPUnit_Framework_Constraint_ExceptionCode(
 938:                             $this->expectedExceptionCode
 939:                         )
 940:                     );
 941:                 }
 942: 
 943:                 return;
 944:             } else {
 945:                 throw $e;
 946:             }
 947:         }
 948: 
 949:         if ($this->expectedException !== null) {
 950:             $this->assertThat(
 951:                 null,
 952:                 new PHPUnit_Framework_Constraint_Exception(
 953:                     $this->expectedException
 954:                 )
 955:             );
 956:         }
 957: 
 958:         return $testResult;
 959:     }
